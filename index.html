async function handleAction(lotId, type) { if (type === 'BUY') { // ถามจำนวนเงินที่ลงทุน (ตั้งต้นไว้ที่ 10,000 หรือตามที่คุณต้อมเคยลง) const capital = prompt("ระบุจำนวนเงินที่ลงทุน (บาท):", "10000"); const buyPrice = prompt("ระบุราคาทองขณะที่ซื้อ (บาทละ):", "42000"); // เก็บไว้เพื่อคำนวณ % เฉยๆ if (capital && buyPrice) { await sendDataToSheet({ action: "BUY", lotId: lotId, capital: parseFloat(capital), buyPrice: parseFloat(buyPrice) }); } } else if (type === 'SELL') { const received = prompt("ระบุจำนวนเงินที่ได้รับทั้งหมดจากการขาย (รวมต้น+กำไร):", ""); if (received) { await sendDataToSheet({ action: "SELL", lotId: lotId, totalReceived: parseFloat(received) }); } } } async function sendDataToSheet(payload) { try { const response = await fetch(WEB_APP_URL, { method: "POST", body: JSON.stringify(payload) }); const result = await response.json(); if (result.status === "success") { alert("บันทึกสำเร็จ!"); location.reload(); } } catch (error) { alert("บันทึกล้มเหลว กรุณาตรวจสอบการ Deploy ใน Apps Script"); } }
