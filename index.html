// แก้ไขฟังก์ชัน handleAction ใน index.html เดิม async function handleAction(lotId, type) { if (type === 'BUY') { const buyPrice = prompt("กรุณาระบุราคาทองที่คุณซื้อจริง (บาทละ):", "40500"); if (buyPrice != null && buyPrice != "") { await sendDataToSheet({ action: "BUY", lotId: lotId, buyPrice: parseFloat(buyPrice) }); } } else if (type === 'SELL') { if (confirm(`ยืนยันการขายถาดที่ ${lotId} และบันทึกกำไร?`)) { await sendDataToSheet({ action: "SELL", lotId: lotId }); } } } // ฟังก์ชันส่งข้อมูลไป Web App URL async function sendDataToSheet(payload) { try { const response = await fetch(WEB_APP_URL, { method: "POST", body: JSON.stringify(payload) }); const result = await response.json(); if (result.status === "success") { alert("บันทึกข้อมูลเรียบร้อย!"); location.reload(); // รีเฟรชหน้าจอเพื่อดูการเปลี่ยนแปลง } else { alert("เกิดข้อผิดพลาด: " + result.message); } } catch (error) { console.error("Error:", error); alert("ไม่สามารถส่งข้อมูลได้ กรุณาตรวจสอบการ Deploy Web App"); } }
